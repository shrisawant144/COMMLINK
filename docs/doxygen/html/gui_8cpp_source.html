1021 void CommLinkGUI::onDataReceived(const DataMessage &msg, const QString &source, const QString &timestamp) {
1022  // Store the message for proper export
1023  receivedMessages.append(msg);
1024 
1025  // Determine protocol based on sender
1026  QString protocol = "Unknown";
1027  QObject *senderObj = sender();
1028 
1029  if (senderObj == tcpClient || senderObj == tcpServer) {
1030  protocol = "TCP";
1031  } else if (senderObj == udpClient || senderObj == udpServer) {
1032  protocol = "UDP";
1033  } else if (senderObj == wsClient || senderObj == wsServer) {
1034  protocol = "WebSocket";
1035  } else if (senderObj == httpClient || senderObj == httpServer) {
1036  protocol = "HTTP";
1037  }
1038 
1039  QString direction = "received";
1040 
1041  QString displayText = msg.toDisplayString();
1042 
1043  QString message = QString("[%1] %2 from %3:\n%4\n")
1044  .arg(timestamp).arg(protocol).arg(source).arg(displayText);
1045 
1046  // Determine if this is from a client or server
1047  bool isClientMessage = (senderObj == tcpClient || senderObj == udpClient ||
1048  senderObj == wsClient || senderObj == httpClient);
1049  bool isServerMessage = (senderObj == tcpServer || senderObj == udpServer ||
1050  senderObj == wsServer || senderObj == httpServer);
1051 
1052  // Add to appropriate tabs
1053  if (isClientMessage) {
1054  clientReceivedEdit->append(message);
1055  logger->logInfo(QString("Client received %1 message from %2").arg(protocol).arg(source));
1056  } else if (isServerMessage) {
1057  serverReceivedEdit->append(message);
1058  logger->logSuccess(QString("Server received %1 message from %2").arg(protocol).arg(source));
1059  }
1060 
1061  // Also add to combined "All Messages" view
1062  receivedEdit->append(message);
1063  logMessage(QString("Received %1 message from %2").arg(source).arg(timestamp), "[RECV] ");
1064 
1065  // Save received message to history
1066  QString host;
1067  int port = 0;
1068 
1069  // Parse host and port based on protocol and source format
1070  if (protocol == "HTTP") {
1071  // For HTTP, source can be a URL (http://host:port) or IP:port [method]
1072  if (source.startsWith("http://") || source.startsWith("https://")) {
1073  // Client side: source is URL like "http://127.0.0.1:5000"
1074  QUrl url(source.split(" ").first()); // Remove any trailing info like [HTTP 200]
1075  host = url.host();
1076  port = url.port(DEFAULT_HTTP_PORT);
1077  } else {
1078  // Server side: source is like "::ffff:127.0.0.1:42966 [POST /]"
1079  QString cleanSource = source.split(" ").first(); // Remove method info
1080  QStringList parts = cleanSource.split(":");
1081  if (parts.size() >= 2) {
1082  // Handle IPv6 format like ::ffff:127.0.0.1:42966
1083  port = parts.last().toInt();
1084  parts.removeLast();
1085  host = parts.join(":");
1086  } else {
1087  host = cleanSource;
1088  port = receivePortEdit->text().toInt();
1089  }
1090  }
1091  } else {
1092  // For TCP/UDP/WebSocket: source format is typically "host:port"
1093  QStringList parts = source.split(":");
1094  if (parts.size() >= 2) {
1095  port = parts.last().toInt();
1096  parts.removeLast();
1097  host = parts.join(":");
1098  } else {
1099  host = source;
1100  port = receivePortEdit->text().toInt();
1101  }
1102  }
1103 
1104  if (!historyManager.saveMessage(direction, protocol, host, port, msg, source)) {
1105  logMessage("Failed to save received message to history", "[WARN] ");
1106  }
