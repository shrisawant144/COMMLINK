140  }

145

148  }
149

152  if (requestLine.size() < 2) {

158  request.path = requestLine[1];

163  int colonIndex = line.indexOf(':');
164  if (colonIndex != -1) {
165  QString key = line.left(colonIndex).trimmed();
166  QString value = line.mid(colonIndex + 1).trimmed();

168  }
169  }
170 
171  // Check for Content-Length
172  int contentLength = 0;
173  if (request.headers.contains("Content-Length")) {
174  contentLength = request.headers["Content-Length"].toInt();
175  }
176 
177  int bodyStart = headerEnd + 4; // After \r\n\r\n
178  if (buffer.size() < bodyStart + contentLength) {
179  return false; // Body not complete yet
180  }
181 
182  // Extract body
183  request.body = buffer.mid(bodyStart, contentLength);
184 
185  // Remove processed request from buffer
186  buffer.remove(0, bodyStart + contentLength);
