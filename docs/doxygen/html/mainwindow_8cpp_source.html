364 {
365  if (connectionPanel->isConnected()) {
366  onDisconnectRequested();
367  return;
368  }
369 
370  if (!validateInputs()) {
371  return;
372  }
373 
374  QString protocol = connectionPanel->getProtocol();
375  QString host = connectionPanel->getHost();
376  int port = connectionPanel->getPort();
377  DataFormatType format = messagePanel->getFormat();
378 
379  if (protocol == "TCP") {
380  tcpClient->setFormat(format);
381  tcpClient->connectToHost(host, static_cast<quint16>(port));
382  logMessage(QString("Connecting to TCP server at %1:%2").arg(host).arg(port), "[CONNECT] ");
383  // Status will be updated when connected() signal is emitted
384  } else if (protocol == "UDP") {
385  udpClient->setFormat(format);
386  if (udpClient->connectToHost(host, static_cast<quint16>(port))) {
387  connectionPanel->setConnectionState(true);
388  logMessage(QString("Connected to UDP server at %1:%2").arg(host).arg(port), "[CONNECT] ");
389  } else {
390  QMessageBox::warning(this, "Connection Error", "Failed to connect via UDP");
391  }
392  } else if (protocol == "WebSocket") {
393  wsClient->setFormat(format);
394  wsClient->connectToServer(host);
395  logMessage(QString("Connecting to WebSocket server at %1").arg(host), "[CONNECT] ");
396  // Status will be updated when connected() signal is emitted
397  } else if (protocol == "HTTP") {
398  httpClient->setFormat(format);
399  httpClient->setConnected(true);
400  connectionPanel->setConnectionState(true);
401  logMessage(QString("HTTP client ready for %1").arg(host), "[CONNECT] ");
402 
403  // Start polling if enabled
404  if (connectionPanel->isHttpPollingEnabled()) {
405  httpClient->startPolling(host, 2000);
406  logMessage("HTTP long-polling enabled", "[HTTP] ");
407  }
408  }

561 // Message handlers
562 void MainWindow::onSendRequested()
563 {
564  QString messageText = messagePanel->getMessage();
565  if (messageText.isEmpty()) {
566  QMessageBox::warning(this, "Error", "Message cannot be empty");
567  return;
568  }
569 
570  DataFormatType format = messagePanel->getFormat();
571  if (!DataMessage::validateInput(messageText, format)) {
572  QMessageBox::warning(this, "Input Error", "Invalid input for selected format");
573  return;
574  }
575 
576  QVariant parsedData = DataMessage::parseInput(messageText, format);
577  DataMessage msg(format, parsedData);
578  QString timestamp = QDateTime::currentDateTime().toString("yyyy-MM-dd HH:mm:ss");
579 
580  QString sendMode = serverPanel->getSendMode();
581 
582  // SERVER-SIDE SENDING
583  if (sendMode.contains("Broadcast") || sendMode.contains("Selected")) {
584  QString serverProto = serverPanel->getProtocol();
585 
586  if (sendMode.contains("Broadcast")) {
587  // BROADCAST TO ALL CLIENTS
588  if (serverProto == "TCP Server" && tcpServer->isListening()) {
589  tcpServer->sendToAll(msg);
590  displayPanel->appendSentMessage(QString("[%1] TCP Server Broadcast:\n%2").arg(timestamp, messageText));
591  logMessage("TCP Server broadcast to all clients", "[SEND] ");
592  historyManager.saveMessage("sent", "TCP", "broadcast", serverPanel->getPort(), msg);
593  } else if (serverProto == "WebSocket Server" && wsServer->isListening()) {
594  wsServer->sendToAll(msg);
595  displayPanel->appendSentMessage(QString("[%1] WebSocket Server Broadcast:\n%2").arg(timestamp, messageText));
596  logMessage("WebSocket Server broadcast to all clients", "[SEND] ");
597  historyManager.saveMessage("sent", "WebSocket", "broadcast", serverPanel->getPort(), msg);
598  } else if (serverProto == "UDP Server" && udpServer->isListening()) {
599  QMessageBox::information(this, "Info", "UDP server replies automatically when it receives messages.");
600  } else if (serverProto == "HTTP Server" && httpServer->isListening()) {
601  httpServer->queueMessageForAll(msg);
602  displayPanel->appendSentMessage(QString("[%1] HTTP Server Broadcast (queued):\n%2").arg(timestamp, messageText));
603  logMessage("HTTP Server queued broadcast", "[SEND] ");
604  historyManager.saveMessage("sent", "HTTP", "broadcast", serverPanel->getPort(), msg);
605  } else {
606  QMessageBox::warning(this, "Error", "Server not listening");
607  }
608  } else {
609  // SEND TO SELECTED CLIENT
610  QString targetClient = serverPanel->getTargetClient();
611  if (targetClient.isEmpty()) {
612  QMessageBox::warning(this, "Error", "No client selected");
613  return;
614  }
615 
616  bool sent = false;
617  if (serverProto == "TCP Server" && tcpServer->isListening()) {
618  QTcpSocket* client = tcpServer->findClientByAddress(targetClient);
619  if (client) {
620  tcpServer->sendToClient(client, msg);
621  sent = true;
622  }
623  } else if (serverProto == "WebSocket Server" && wsServer->isListening()) {
624  QWebSocket* client = wsServer->findClientByAddress(targetClient);
625  if (client) {
626  wsServer->sendToClient(client, msg);
627  sent = true;
628  }
629  } else if (serverProto == "UDP Server" && udpServer->isListening()) {
630  QStringList parts = targetClient.split(":");
631  if (parts.size() == 2) {
632  QHostAddress addr(parts[0]);
633  quint16 port = parts[1].toUShort();
634  udpServer->sendTo(addr, port, msg);
635  sent = true;
636  }
637  } else if (serverProto == "HTTP Server" && httpServer->isListening()) {
638  QTcpSocket* client = httpServer->findClientByAddress(targetClient);
639  if (client) {
640  httpServer->queueMessageForClient(client, msg);
641  sent = true;
642  }
643  }
644 
645  if (sent) {
646  displayPanel->appendSentMessage(QString("[%1] %2 to %3:\n%4").arg(timestamp, serverProto, targetClient, messageText));
647  logMessage(QString("%1 sent to %2").arg(serverProto, targetClient), "[SEND] ");
648  // Strip " Server" suffix for database (CHECK constraint requires TCP/UDP/WebSocket/HTTP only)
649  QString dbProtocol = serverProto;
650  dbProtocol.remove(" Server");
651  historyManager.saveMessage("sent", dbProtocol, targetClient, serverPanel->getPort(), msg);
652  } else {
653  QMessageBox::warning(this, "Error", "Client not found or disconnected");
654  }
655  }
656  return;
657  }
658 
659  // CLIENT-SIDE SENDING
660  QString proto = connectionPanel->getProtocol();
661 
662  if (proto == "HTTP") {
663  QString url = connectionPanel->getHost();
664  if (!url.startsWith("http://") && !url.startsWith("https://")) {
665  url = "http://" + url;
666  }
667 
668  QString methodStr = connectionPanel->getHttpMethod();
669  HttpClient::Method method = HttpClient::POST;
670  if (methodStr == "GET") method = HttpClient::GET;

850 // Network event handlers
851 void MainWindow::onDataReceived(const DataMessage &msg, const QString &source, const QString &timestamp)
852 {
853  receivedMessages.append(msg);
854 
855  // Determine protocol based on sender
856  QString protocol = "Unknown";
857  QObject *senderObj = sender();
858 
859  if (senderObj == tcpClient || senderObj == tcpServer) {
860  protocol = "TCP";
861  } else if (senderObj == udpClient || senderObj == udpServer) {
862  protocol = "UDP";
863  } else if (senderObj == wsClient || senderObj == wsServer) {
864  protocol = "WebSocket";
865  } else if (senderObj == httpClient || senderObj == httpServer) {
866  protocol = "HTTP";
867  }
868 
869  QString displayText = msg.toDisplayString();
870  QString message = QString("[%1] %2 from %3:\n%4\n")
871  .arg(timestamp, protocol, source, displayText);
872 
873  // Determine if this is from a client or server
874  bool isClientMessage = (senderObj == tcpClient || senderObj == udpClient ||
875  senderObj == wsClient || senderObj == httpClient);
876  bool isServerMessage = (senderObj == tcpServer || senderObj == udpServer ||
877  senderObj == wsServer || senderObj == httpServer);
878 
879  // Add to appropriate tabs
880  if (isClientMessage) {
881  displayPanel->appendReceivedMessage(message, false);
882  logMessage(QString("Client received %1 message from %2").arg(protocol, source), "[RECV] ");
883  } else if (isServerMessage) {
884  displayPanel->appendReceivedMessage(message, true);
885  logMessage(QString("Server received %1 message from %2").arg(protocol, source), "[RECV] ");
886  }
887 
888  // Save to history database
889  QString host = source;
890  int port = 0;
891 
892  // Parse host and port from source
893  if (protocol == "HTTP") {
894  if (source.startsWith("http://") || source.startsWith("https://")) {
895  QUrl url(source.split(" ").first());
896  host = url.host();
897  port = url.port(80);
898  } else {
899  QStringList parts = source.split(" ").first().split(":");
900  if (parts.size() >= 2) {
901  port = parts.last().toInt();
902  parts.removeLast();
903  host = parts.join(":");
904  }
905  }
906  } else {
907  QStringList parts = source.split(":");
908  if (parts.size() >= 2) {
909  port = parts.last().toInt();
910  parts.removeLast();
911  host = parts.join(":");
912  }
913  }
914 
915  if (!historyManager.saveMessage("received", protocol, host, port, msg, source)) {
916  logMessage("Failed to save message to history", "[WARN] ");
917  }
918 
919  // Save settings on successful receive
920  saveSettings();
921 }
